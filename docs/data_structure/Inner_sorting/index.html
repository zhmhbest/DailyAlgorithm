插入排序

算法 完全随机 基本有序 逆排序 少有不同
插入排序
  • 稳定
  • O(1)O(1)O(1)额外空间
  • O(n2)O(n^2)O(n2)比较和互换
  • 自适应性——当基本有序时，耗费O(n)O(n)O(n)的时间
  • 非常低的开销
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include "array.inl"

/**
 * 直接插入排序
 */
void insertSort(int arr[], int len)
{
    // 从第二位开始最后一位
    for (int i = 1; i < len; i++)
    {
        // “当前元素”左侧数据已经有序

        // 1.“当前元素”不需要移动
        int j = i - 1;
        if (arr[i] >= arr[j])
            continue;

        // 2.将“当前元素”插入到左侧数据，使之依然有序
        int x = arr[i];
        for (; j >= 0 && x < arr[j]; j--)
            arr[j + 1] = arr[j];
        arr[j + 1] = x;
    }
}

int main()
{
    // int arr[] = {2, 4, 5, 1, 8};
    int arr[] = {8, 5, 4, 2, 1};
    const int len = 5;
    printArray(arr, len);
    insertSort(arr, len);
    printArray(arr, len);
    return 0;
}

选择排序

在交换单位的消耗成本很高的应用中，选择排序很可能也是合适的算法选择。

算法 完全随机 基本有序 逆排序 少有不同
选择排序
  • 不稳定
  • O(1)O(1)O(1) 额外空间
  • O(n2)O(n2)O(n2) 比较
  • O(n)O(n)O(n)互换
  • 不适应性
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include "array.inl"

/**
 * 简单选择排序
 */
void selectSort(int arr[], int len)
{
    for (int i = 0; i < len; i++)
    {
        int j = minKey(arr, len, i);
        if (i != j)
            swapValue(arr, i, j);
    }
}

int main()
{
    // int arr[] = {2, 4, 5, 1, 8};
    int arr[] = {8, 5, 4, 2, 1};
    const int len = 5;
    printArray(arr, len);
    selectSort(arr, len);
    printArray(arr, len);
    return 0;
}

冒泡排序

算法 完全随机 基本有序 逆排序 少有不同
冒泡排序
  • 稳定
  • O(1)O(1)O(1)额外空间
  • O(n2)O(n^2)O(n2)比较和互换
  • 适应性——当接近排序完的时候耗费 O(n) 时间
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include "array.inl"

/**
 * 冒泡排序
 */
void bubbleSort(int arr[], int len)
{
    // [ →i→ ]
    for (int i = 0; i < len; i++)
    {
        // (i ←j← ]
        for (int j = len - 1; j > i; j--)
        {
            if (arr[j - 1] > arr[j])
            {
                swapValue(arr, j - 1, j);
            }
        }
    }
}

int main()
{
    // int arr[] = {2, 4, 5, 1, 8};
    int arr[] = {8, 5, 4, 2, 1};
    const int len = 5;
    printArray(arr, len);
    bubbleSort(arr, len);
    printArray(arr, len);
    return 0;
}

希尔排序

算法 完全随机 基本有序 逆排序 少有不同
希尔排序
  • 不稳定
  • O(1)O(1)O(1)额外空间
  • 需要O(n32)O(n^{\frac{3}{2}})O(n23​)时间
  • 适应性——当接近排序完的时候耗费O(n⋅log⁡2(n))O(n \cdot \log_2(n))O(n⋅log2​(n))时间
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include "array.inl"

void shellSortGroupTest(int arr[], int len)
{
    // 将每组分为2,4,8...个
    for (int step = len / 2; step > 0; step /= 2)
    {
        printf("step=%d\n", step);
        for (int i = step; i < len; i++)
        {
            int j = i,
                k = j - step;
            printf("group=(%d", arr[j]);
            while (k >= 0)
            {
                j = k;
                k = j - step;
                printf(", %d", arr[j]);
            }
            printf(")\n");
        }
    }
}

/**
 * 希尔排序
 */
void shellSort(int arr[], int len)
{
    for (int step = len / 2; step > 0; step /= 2)
    {
        for (int i = step; i < len; i++)
        {
            int j = i,
                k = j - step,
                dump = arr[j];
            // 插入排序
            while (k >= 0 && arr[k] > dump)
            {
                arr[j] = arr[k];
                j = k;
                k = j - step;
            }
            arr[j] = dump;
        }
    }
}

int main()
{
    int arr[] = {2, 5, 1, 8, 23, 546, 76, 34, 82, 3, 6, 4};
    const int len = 12;
    printArray(arr, len);
    shellSortGroupTest(arr, len);
    shellSort(arr, len);
    printArray(arr, len);
    return 0;
}
